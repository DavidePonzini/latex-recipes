\begin{recipe}[label=risoJawad]{Riso e pastina allo zafferano}
    \begin{header}
        \portion{2}[piatti]
        \source{Jawad Shurrush}
        \recipedate{Febbraio 2024}

        \preparationTime     {\timeM{2}}
        \cookingTime         {\timeM{\tilde{}2}}{Fiamma alta}
        \cookingTime         {\timeM{13}}{Fiamma medio-bassa}
        \preparationWait     {\timeM{2-3}}
    \end{header}

    \begin{ingredients}
        \ingredient[150][g]{Riso Basmati}
        \ingredient[50-70][g]{Pastina corta}
        \ingredient[220-230][g]{Pastina corta}
        \ingredient[40][g]{Olio d'oliva}
        \ingredient{Zafferano}
        \ingredient{Noce moscata}
        \ingredient{Cannella}
        \ingredient[\textit{(Poco)}]{Pepe inglese}
        \ingredient{Pepe nero}
        \ingredient{Sale}
    \end{ingredients}

    \begin{preparation}
        \step Mettere in una pentola antiaderente olio e pastina.
        \step Far rosolare fino a che la pasta diventa ben dorata.
        \step Aggiungere riso e far scottare.
        \step Aggiungere acqua e spezie, far bollire l'acqua.
        \step Cuocere col coperchio (o piatto, deve andare un po' sotto pressione).
        \step Lasciare riposare.
        \step Mescolare delicatamente dal basso verso l'alto.
    \end{preparation}

    \begin{suggestion}
        \suggestionMark $Pasta = \sfrac{1}{3} * Riso$
        \suggestionMark $Acqua = 2 * Riso + Pasta$
    \end{suggestion}
\end{recipe}
